\[
\Delta H_R^{\circ}(T_2) = \Delta H_R^{\circ}(T_1) + \Delta C_p \cdot (T_2 - T_1),

\[
\Delta C_p = \sum C_{p,m}^{\circ}(\text{products}) - \sum C_{p,m}^{\circ}(\text{reactants}).
