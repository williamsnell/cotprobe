\[
\text{H}_2\text{SO}_4 + 2 \, \text{NaCN} \rightarrow \text{Na}_2\text{SO}_4 + 2 \, \text{HCN}
\]

\text{Molar mass of NaCN} = 23 \, (\text{Na}) + 12 \, (\text{C}) + 14 \, (\text{N}) = 49 \, \text{g/mol}

\text{Moles of NaCN} = \frac{\text{Mass}}{\text{Molar mass}} = \frac{196}{49} = 4 \, \text{mol}
