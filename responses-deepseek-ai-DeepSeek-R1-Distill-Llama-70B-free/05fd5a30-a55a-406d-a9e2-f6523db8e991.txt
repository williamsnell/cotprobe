\[

\]

\[
\text{Multiplier} = \frac{1}{1 - \text{MPC}} = \frac{1}{1 - 0.8} = \frac{1}{0.2} = 5.
\]

\[
\text{Increase in Government Expenditures} = \frac{\text{GDP Gap}}{\text{Multiplier}} = \frac{100 \text{ billion}}{5} = 20 \text{ billion}.
\]
