  \[
  1 \, \text{Torr} = 133.322 \, \text{Pa}, \quad 200 \, \text{Torr} = 200 \times 133.322 = 26,664 \, \text{Pa}.

  \[
  1 \, \text{dm}^3 = 0.001 \, \text{m}^3, \quad 3.3 \, \text{dm}^3 = 3.3 \times 0.001 = 0.0033 \, \text{m}^3.
  \]

\[
w = -P_{\text{ext}} \Delta V = -(26,664 \, \text{Pa}) \times (0.0033 \, \text{m}^3).
