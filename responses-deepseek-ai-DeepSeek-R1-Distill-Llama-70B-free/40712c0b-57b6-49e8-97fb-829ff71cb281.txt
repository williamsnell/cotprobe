\[
\text{Dividend Yield} = \frac{\text{Annual Dividends Per Share}}{\text{Price Per Share}}

\[
\text{Dividend Yield} = \frac{2}{60} = \frac{1}{30} \approx 0.0333 \, \text{or} \, 3\frac{1}{3}\%.
