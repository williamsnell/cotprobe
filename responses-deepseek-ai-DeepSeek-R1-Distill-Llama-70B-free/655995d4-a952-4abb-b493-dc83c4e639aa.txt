\[
P(AABB) = P(AA) \cdot P(BB) = \frac{1}{4} \cdot \frac{1}{4} = \frac{1}{16}
\]

\[
P(aabb) = P(aa) \cdot P(bb) = \frac{1}{4} \cdot \frac{1}{4} = \frac{1}{16}

\[
