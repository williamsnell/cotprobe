\[
E_{\text{photon}} = \frac{(6.626 \times 10^{-34})(3.0 \times 10^8)}{3.0 \times 10^{-7}} = 6.626 \times 10^{-19} \, \text{J}

\[
K.E. = E_{\text{photon}} - \phi = 6.626 \times 10^{-19} - 3.204 \times 10^{-19} = 3.422 \times 10^{-19} \, \text{J}

K.E. = \frac{1}{2} m v^2 \quad \Rightarrow \quad v = \sqrt{\frac{2 K.E.}{m}}
