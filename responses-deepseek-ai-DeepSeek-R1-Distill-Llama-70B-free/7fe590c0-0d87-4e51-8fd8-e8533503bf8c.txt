\[
\Delta H_{\text{sol}} = \left[ \Delta H_f(\mathrm{Ag}^+, \mathrm{aq}) + \Delta H_f(\mathrm{Cl}^-, \mathrm{aq}) \right] - \Delta H_f(\mathrm{AgCl}, \mathrm{s})
