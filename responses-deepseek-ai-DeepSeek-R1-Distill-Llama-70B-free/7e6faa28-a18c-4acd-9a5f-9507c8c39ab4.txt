W = \mathbf{F} \cdot \mathbf{d}
\]

\[
\mathbf{d} = (x_{\text{final}} - x_{\text{initial}}, y_{\text{final}} - y_{\text{initial}}, z_{\text{final}} - z_{\text{initial}})
