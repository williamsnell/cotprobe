\[
k = \frac{F}{x} = \frac{6.0}{0.030} = 200 \, \text{N/m}.
\]

\[
T = 2\pi \sqrt{\frac{m}{k}},

\[
\frac{m}{k} = \frac{0.50}{200} = 0.0025.

\[
\sqrt{0.0025} = 0.05.
