   \[
   \left(\frac{z_{\alpha}}{z_p}\right)^2 = \left(\frac{2}{1}\right)^2 = 4.

   \[
   v_{\alpha} = \frac{v_p}{2}.

   \[
   \left(\frac{v_p}{v_{\alpha}}\right)^2 = \left(\frac{v_p}{v_p/2}\right)^2 = 4.
   \]

   \[
   \frac{\left(\frac{z_{\alpha}^2}{v_{\alpha}^2}\right)}{\left(\frac{z_p^2}{v_p^2}\right)} = \left(\frac{z_{\alpha}^2}{z_p^2}\right) \cdot \left(\frac{v_p^2}{v_{\alpha}^2}\right) = 4 \cdot 4 = 16.
