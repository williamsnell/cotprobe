\[
\beta_{\text{glass}} = 3 \alpha_{\text{glass}} = 3 \times 0.40 \times 10^{-5} = 1.2 \times 10^{-5} \, \text{K}^{-1}.

\[
\Delta V_{\text{glass}} = V_0 \cdot \beta_{\text{glass}} \cdot \Delta T = 200 \cdot 1.2 \times 10^{-5} \cdot 80 = 0.192 \, \text{cm}^3.
