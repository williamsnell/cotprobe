\[
6\left(j^2 - \frac{2}{3}j + \frac{1}{9} - \frac{1}{9}\right) + 12.
\]

\[
6\left(\left(j - \frac{1}{3}\right)^2 - \frac{1}{9}\right) + 12.

\[
6\left(j - \frac{1}{3}\right)^2 - 6 \cdot \frac{1}{9} + 12 = 6\left(j - \frac{1}{3}\right)^2 - \frac{2}{3} + 12.

\[
6\left(j - \frac{1}{3}\right)^2 + \frac{34}{3}.
\]
