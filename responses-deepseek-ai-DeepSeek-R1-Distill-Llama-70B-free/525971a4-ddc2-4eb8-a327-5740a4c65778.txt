\[
n = \frac{\text{mass}}{\text{molar mass}} = \frac{92400 \, \mathrm{g}}{28 \, \mathrm{g/mol}} = 3300 \, \mathrm{mol}

\[
\left(P + \frac{a n^2}{V^2}\right) \left(\frac{V}{n} - b\right) = RT
