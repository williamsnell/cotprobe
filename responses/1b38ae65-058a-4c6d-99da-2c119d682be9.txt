\[
\Gamma = \frac{Z_{\text{load}} - Z_{\text{line}}}{Z_{\text{load}} + Z_{\text{line}}} = \frac{75 - 50}{75 + 50} = \frac{25}{125} = 0.2
\]

\[
P_{\text{reflected}} = \Gamma^2 \cdot P_{\text{incident}} = (0.2)^2 \cdot 50\% = 4\% \cdot P_{\text{incident}}
