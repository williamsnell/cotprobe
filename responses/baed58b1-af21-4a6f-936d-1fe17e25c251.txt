\[
\ln(1,000,000) \approx 13.8155
\]

\[
E[U] = p \cdot \ln(1,100,000) + (1-p) \cdot \ln(900,000)

\ln(1,100,000) \approx 13.9108, \quad \ln(900,000) \approx 13.7101
