\[
\text{SnO}_2 = 118.71 \, \text{g/mol (Sn)} + 2 \times 16.00 \, \text{g/mol (O)} = 150.71 \, \text{g/mol}.
\]

\[
\text{Moles of SnO}_2 = \frac{\text{Mass of SnO}_2}{\text{Molar mass of SnO}_2} = \frac{0.600 \, \text{g}}{150.71 \, \text{g/mol}} \approx 0.003983 \, \text{mol}.
\]

\[
\text{Moles of Sn} = 0.003983 \, \text{mol}.
\]

\[
\text{Mass of Sn} = \text{Moles of Sn} \times \text{Molar mass of Sn} = 0.003983 \, \text{mol} \times 118.71 \, \text{g/mol} \approx 0.4728 \, \text{g}.
\]
