\[
\text{Volume} = 2 \, \text{cm} \times 2 \, \text{cm} \times 10 \, \text{cm} = 40 \, \text{cm}^3
\]

\[
\text{Density} = \frac{\text{Mass}}{\text{Volume}} = \frac{320 \, \text{g}}{40 \, \text{cm}^3} = 8 \, \text{g/cm}^3
\]
