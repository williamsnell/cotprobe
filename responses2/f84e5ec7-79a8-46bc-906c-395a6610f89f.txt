   \text{Volume} = \text{Cross-sectional area} \times \text{Change in draft} = 3,000 \, \text{m}^2 \times 0.2 \, \text{m} = 600 \, \text{m}^3

   \[
   \text{Weight} = \text{Density} \times \text{Volume} = 10 \, \text{kN/m}^3 \times 600 \, \text{m}^3 = 6,000 \, \text{kN}
   \]
