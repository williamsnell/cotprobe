\[
\lambda = 4h = 4 \times 2 = 8 \, \text{m}.

\[
f = \frac{c}{\lambda},

\[
f = \frac{3 \times 10^8}{8} = 37.5 \times 10^6 \, \text{Hz} = 37.5 \, \text{MHz}.
\]
