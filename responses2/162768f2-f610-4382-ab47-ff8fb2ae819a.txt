\[
\Delta G = -85.40 + 36.5T

\[
\Delta G = \Delta H - T\Delta S
\]

\[
\frac{d\Delta G}{dT} = 36.5 \, \text{J/K}
\]

\[
\frac{d\Delta G}{dT} = \frac{d\Delta H}{dT} - \Delta S

\[
36.5 = 36.5 - \Delta S
