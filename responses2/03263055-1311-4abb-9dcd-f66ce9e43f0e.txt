\[
\frac{1}{C_{\text{eq}}} = \frac{1}{C_1} + \frac{1}{C_2} + \dots + \frac{1}{C_n}
\]

\[
\frac{1}{C_{\text{eq}}} = 5 \times \frac{1}{5} = 1 \quad \Rightarrow \quad C_{\text{eq}} = 1 \, \mu\text{F}.
\]
