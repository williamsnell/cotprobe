\[
g = \frac{G M}{r^2}
\]

\[
g_0 = \frac{G M}{R^2}
\]

\[
\frac{G M}{r^2} = \frac{1}{2} \cdot \frac{G M}{R^2}
\]

\[
\frac{1}{r^2} = \frac{1}{2 R^2} \implies r^2 = 2 R^2 \implies r = R \cdot \sqrt{2}
